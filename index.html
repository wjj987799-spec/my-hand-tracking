// 在代码中找到之前的 // 绘制粒子 部分，用这段逻辑替换： let particles = []; function updateParticles(targetX, targetY) { // 每次动作时产生粒子 for (let i = 0; i < 3; i++) { particles.push({ x: targetX, y: targetY, vx: (Math.random() - 0.5) * 10, vy: (Math.random() - 0.5) * 10, life: 1.0, color: Math.random() * 360 }); } } // 核心渲染循环 function track() { // ... (保持之前的差分追踪逻辑) ... // 渲染特效 ctx.fillStyle = 'rgba(0, 0, 0, 0.15)'; // 产生拖尾效果的关键 ctx.fillRect(0, 0, c.width, c.height); updateParticles(c.width - x, y); // 把追踪到的位置传入 particles.forEach((p, index) => { p.x += p.vx; p.y += p.vy; p.life -= 0.02; // 粒子生命周期 ctx.fillStyle = `hsla(${p.color}, 80%, 60%, ${p.life})`; ctx.beginPath(); ctx.arc(p.x, p.y, p.life * 5, 0, Math.PI * 2); ctx.fill(); if (p.life <= 0) particles.splice(index, 1); }); requestAnimationFrame(track); }
